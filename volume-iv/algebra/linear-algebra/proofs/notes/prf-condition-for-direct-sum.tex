\begin{theorem*}[Condition for a Direct Sum]
Let $V = U_1+\cdots+U_m$ where each $U_j$ is a subspace of $V$.
Then $V = U_1\oplus\cdots\oplus U_m$ if and only if
$u_1+\cdots+u_m = 0$ with $u_j \in U_j$ implies $u_1 = \cdots = u_m = 0$.
\end{theorem*}

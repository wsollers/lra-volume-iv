\begin{theorem*}[Characterizations of Order-Isomorphisms for Finite Posets]
Let $(P, \leq_P)$ and $(Q, \leq_Q)$ be finite partially ordered sets and
let $\varphi : P \to Q$ be a bijection.  The following are equivalent:
(i) $\varphi$ is an order-isomorphism;
(ii) $x <_P y \Leftrightarrow \varphi(x) <_Q \varphi(y)$ for all $x,y \in P$;
(iii) $x \prec_P y \Leftrightarrow \varphi(x) \prec_Q \varphi(y)$ for all $x,y \in P$.
\end{theorem*}

\begin{proof}[Proof]
\textbf{(i) $\Leftrightarrow$ (ii).}
The equivalence follows directly from the definitions: $\varphi$ is an
order-isomorphism if and only if $x \leq_P y \Leftrightarrow \varphi(x)
\leq_Q \varphi(y)$.  Since $\varphi$ is bijective, $x = y$ if and only if
$\varphi(x) = \varphi(y)$, so $x <_P y$ ($x \leq_P y$ and $x \neq y$)
if and only if $\varphi(x) <_Q \varphi(y)$.

\textbf{(ii) $\Rightarrow$ (iii).}
Assume (ii).  Suppose $x \prec_P y$, so $x <_P y$ and there is no $z$
with $x <_P z <_P y$.  By (ii), $\varphi(x) <_Q \varphi(y)$.  If there
existed $w \in Q$ with $\varphi(x) <_Q w <_Q \varphi(y)$, then since
$\varphi$ is bijective, $w = \varphi(u)$ for some $u \in P$, and (ii)
gives $x <_P u <_P y$, contradicting $x \prec_P y$.  Therefore
$\varphi(x) \prec_Q \varphi(y)$.  The reverse implication is symmetric.

\textbf{(iii) $\Rightarrow$ (ii).}
Assume (iii).  Suppose $x <_P y$.  Since $P$ is finite, there exists a
covering chain $x = x_0 \prec_P x_1 \prec_P \cdots \prec_P x_n = y$.
By (iii), $\varphi(x_0) \prec_Q \cdots \prec_Q \varphi(x_n)$, so
$\varphi(x) <_Q \varphi(y)$.  The reverse implication follows by
bijectivity of $\varphi$.
\end{proof}

\begin{remark*}[Proof structure]
A three-way cycle of implications.  The key step is (ii) $\Rightarrow$
(iii): a covering is an irreducible strict comparison, and if a
covering is reflected by a covering then there can be no intermediate
element in $Q$ (by bijectivity carrying it back to $P$).
\end{remark*}

\begin{remark*}[Dependencies]
\hyperref[def:order-isomorphism]{Order-isomorphism},
\hyperref[def:covering-relation]{covering relation},
finiteness of $P$ (used in (iii) $\Rightarrow$ (ii) to decompose $<$ into a covering chain).
\end{remark*}

\begin{definitionbox}{Definition (Ordered Field)}
\begin{definition}[Ordered Field]\label{def:ordered-field}
An ordered field is a field $F$ together with a linear order $\leq$ such that
for all $a,b,c\in F$,
\[
a\leq b \Rightarrow a+c\leq b+c,
\]
and
\[
0\leq a,\ 0\leq b \Rightarrow 0\leq ab.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
\operatorname{IsOrderedField}(F,\leq)
\Longleftrightarrow\;&
F\text{ is a field}
\land
\leq\text{ is a linear order on }F
\\
&{}\land
\forall a,b,c\in F{:}\; a\leq b \Longrightarrow a+c\leq b+c
\\
&{}\land
\forall a,b\in F{:}\; 0\leq a\land 0\leq b \Longrightarrow 0\leq ab.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{IsOrderedField}(F,\leq)
\]
means that \(F\) is a field whose order is compatible with translation and
with multiplication by nonnegative elements.
\end{remark*}

\begin{remark*}[Interpretation]
An ordered field is the package needed to use algebraic operations and order
inequalities together.
\end{remark*}

\begin{theorembox}{Theorem (Order Is Translation Invariant)}
\begin{theorem}[Order Is Translation Invariant]\label{thm:ordered-field-translation-invariant}
In an ordered field, if
\[
a<b,
\]
then for every $c$,
\[
a+c<b+c.
\]
\hyperref[prf:ordered-field-translation-invariant]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsOrderedField}(F,\leq)
\Longrightarrow
\forall a,b,c\in F{:}\; a<b \Longrightarrow a+c<b+c.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
a<b
\Longrightarrow
a+c<b+c
\]
means that translating both sides of a strict inequality by the same field
element preserves the inequality.
\end{remark*}

\begin{remark*}[Interpretation]
Order comparisons in an ordered field do not depend on where the origin is
placed.
\end{remark*}

\begin{theorembox}{Theorem (Order Is Preserved by Positive Multiplication)}
\begin{theorem}[Order Is Preserved by Positive Multiplication]\label{thm:ordered-field-positive-multiplication-preserves-order}
In an ordered field, if
\[
a<b
\quad\text{and}\quad
c>0,
\]
then
\[
ac<bc.
\]
\hyperref[prf:ordered-field-positive-multiplication-preserves-order]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsOrderedField}(F,\leq)
\Longrightarrow
\forall a,b,c\in F{:}\; a<b\land c>0 \Longrightarrow ac<bc.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
a<b\land c>0
\Longrightarrow
ac<bc
\]
means that multiplying both sides by a positive field element preserves strict
order.
\end{remark*}

\begin{remark*}[Interpretation]
Positive scaling changes size but not order direction.
\end{remark*}

\begin{theorembox}{Theorem (Squares Are Nonnegative)}
\begin{theorem}[Squares Are Nonnegative]\label{thm:ordered-field-squares-nonnegative}
In an ordered field, for every $x$,
\[
x^2\ge 0.
\]
\hyperref[prf:ordered-field-squares-nonnegative]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsOrderedField}(F,\leq)
\Longrightarrow
\forall x\in F{:}\; x^2\geq 0.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
x\in F
\Longrightarrow
x^2\geq 0
\]
means that every square in an ordered field is nonnegative.
\end{remark*}

\begin{remark*}[Interpretation]
The order structure forces products of an element with itself to be
nonnegative.
\end{remark*}

\begin{theorembox}{Theorem (No Ordered Field Has Square Root of Negative One)}
\begin{theorem}[No Ordered Field Has Square Root of Negative One]\label{thm:ordered-field-no-square-root-negative-one}
In an ordered field, there is no $x$ such that
\[
x^2=-1.
\]
\hyperref[prf:ordered-field-no-square-root-negative-one]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsOrderedField}(F,\leq)
\Longrightarrow
\neg\exists x\in F{:}\; x^2=-1_F.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\neg\exists x\in F{:}\; x^2=-1_F
\]
means that \(-1\) has no square root in an ordered field.
\end{remark*}

\begin{remark*}[Interpretation]
Since squares are nonnegative and \(1\) is positive, \(-1\) cannot be a square.
\end{remark*}

\begin{lemmabox}{Lemma (Product Sign Laws in an Ordered Field)}
\begin{lemma}[Product Sign Laws in an Ordered Field]\label{lem:ordered-field-product-sign-laws}
In an ordered field, the usual product sign laws hold:
\[
(+)(+)=(+),\qquad (-)(-)=(+),\qquad (+)(-)=(-),\qquad (-)(+) = (-).
\]
\hyperref[prf:ordered-field-product-sign-laws]{Go to proof.}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
\operatorname{IsOrderedField}(F,\leq)
\Longrightarrow
\forall a,b\in F{:}\;&
(a>0\land b>0 \Longrightarrow ab>0)
\\
&{}\land
(a<0\land b<0 \Longrightarrow ab>0)
\\
&{}\land
(a>0\land b<0 \Longrightarrow ab<0)
\\
&{}\land
(a<0\land b>0 \Longrightarrow ab<0).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{sgn}(ab)=\operatorname{sgn}(a)\operatorname{sgn}(b)
\]
means that multiplication follows the usual positive and negative sign rules.
\end{remark*}

\begin{remark*}[Interpretation]
Products are positive when the factors have the same sign and negative when
the factors have opposite signs.
\end{remark*}

\begin{lemmabox}{Lemma (Division Order Laws in an Ordered Field)}
\begin{lemma}[Division Order Laws in an Ordered Field]\label{lem:ordered-field-division-order-laws}
In an ordered field, division by a positive element preserves order and
division by a negative element reverses order.
\hyperref[prf:ordered-field-division-order-laws]{Go to proof.}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
\operatorname{IsOrderedField}(F,\leq)
\Longrightarrow
\forall a,b,c\in F{:}\;&
c>0 \Longrightarrow
\left(a<b \Longleftrightarrow \frac{a}{c}<\frac{b}{c}\right)
\\
&{}\land
c<0 \Longrightarrow
\left(a<b \Longleftrightarrow \frac{b}{c}<\frac{a}{c}\right).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
c>0\text{ preserves division order},
\qquad
c<0\text{ reverses division order}.
\]
means that dividing an inequality by a positive element preserves its direction,
while dividing by a negative element reverses its direction.
\end{remark*}

\begin{remark*}[Interpretation]
Division inherits its order behavior from multiplication by reciprocals.
\end{remark*}

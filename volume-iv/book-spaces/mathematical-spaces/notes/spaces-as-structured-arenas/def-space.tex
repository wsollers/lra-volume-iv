\begin{definitionbox}{Definition --- Space}
\begin{definition}[Space]\label{def:mathematical-space}
A \textbf{space} is a nonempty set.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  X\ \text{is a space}
  \;\Longleftrightarrow\;
  X \neq \varnothing.
\]
\end{remark*}

\begin{remark*}[Interpretation]
At the base level, the word ``space'' records that there is an arena of
points under discussion.  No distance, topology, operation, or measure has
yet been chosen.  Later phrases such as metric space, topological space, and
measure space add structure to this underlying nonempty set.
\end{remark*}

\begin{remark*}[Historical note]
This convention appears as Definition~2.1 in
Kainth's \emph{A Comprehensive Textbook on Metric Spaces}
\cite{KainthComprehensiveMetricSpaces2023}.
\end{remark*}

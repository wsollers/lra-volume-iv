\subsection{Closure Under Countable Unions and Intersections}

\begin{remark*}[Exposition]
Countable closure is the analytic strengthening of finite closure.  It allows
a family to absorb the result of combining an entire sequence of admitted
subsets.
\end{remark*}

\begin{remark*}[Exposition]
Countable union closure and countable intersection closure are distinct
requirements unless complement closure is also available.  With complement
closure, De Morgan's laws turn one countable condition into the other.
\end{remark*}

\begin{definition}[Closed Under Countable Unions]\label{def:closed-under-countable-unions}
Let \(X\) be a set and let \(\mathcal{F}\subseteq\mathcal{P}(X)\).  The family
\(\mathcal{F}\) is \emph{closed under countable unions} if for every sequence
\(\{A_n\}_{n\in\mathbb{N}}\) of members of \(\mathcal{F}\), one has
\[
  \bigcup_{n=1}^{\infty} A_n \in \mathcal{F}.
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  \forall \{A_n\}_{n\in\mathbb{N}}\,
  \bigl((\forall n,\ A_n\in\mathcal{F})
  \Rightarrow
  \bigcup_{n=1}^{\infty}A_n\in\mathcal{F}\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&C_{\sigma\cup}
=\mathsf{ClosureContext}(X,\mathcal{F}^{\mathbb{N}},\mathcal{F},\bigcup_{\sigma}),\\
&\operatorname{ClosedUnder}(\bigcup_{\sigma},\mathcal{F},C_{\sigma\cup}).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Closure under countable unions says that a countable accumulation of admitted
alternatives is still admitted.
\end{remark*}

\begin{definition}[Closed Under Countable Intersections]\label{def:closed-under-countable-intersections}
Let \(X\) be a set and let \(\mathcal{F}\subseteq\mathcal{P}(X)\).  The family
\(\mathcal{F}\) is \emph{closed under countable intersections} if for every
sequence \(\{A_n\}_{n\in\mathbb{N}}\) of members of \(\mathcal{F}\), one has
\[
  \bigcap_{n=1}^{\infty} A_n \in \mathcal{F}.
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  \forall \{A_n\}_{n\in\mathbb{N}}\,
  \bigl((\forall n,\ A_n\in\mathcal{F})
  \Rightarrow
  \bigcap_{n=1}^{\infty}A_n\in\mathcal{F}\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&C_{\sigma\cap}
=\mathsf{ClosureContext}(X,\mathcal{F}^{\mathbb{N}},\mathcal{F},\bigcap_{\sigma}),\\
&\operatorname{ClosedUnder}(\bigcap_{\sigma},\mathcal{F},C_{\sigma\cap}).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Closure under countable intersections says that a countable accumulation of
admitted simultaneous conditions is still admitted.
\end{remark*}

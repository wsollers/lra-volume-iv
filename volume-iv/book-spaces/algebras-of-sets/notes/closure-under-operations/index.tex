\section{Closure Under Operations}

\begin{toolkitbox}{Toolkit --- Closure Under Operations}
\textbf{Concept family:} a family of subsets remains internally stable when
specified operations are applied to its members.

\smallskip
\begin{center}
\small
\renewcommand{\arraystretch}{1.15}
\begin{tabularx}{\linewidth}{>{\raggedright\arraybackslash}p{0.34\linewidth}
                            >{\raggedright\arraybackslash}X}
\toprule
\textbf{Closure condition} & \textbf{Meaning} \\
\midrule
\hyperref[def:closed-under-operation]{Closed under an operation} &
The operation sends admitted inputs to admitted outputs. \\
\hyperref[def:closed-under-finite-unions]{Closed under finite unions} &
Finite unions of admitted subsets are admitted. \\
\hyperref[def:closed-under-finite-intersections]{Closed under finite intersections} &
Finite intersections of admitted subsets are admitted. \\
\hyperref[def:closed-under-complements]{Closed under complements} &
Complements relative to the ambient set are admitted. \\
\hyperref[def:closed-under-countable-unions]{Closed under countable unions} &
Unions of sequences of admitted subsets are admitted. \\
\hyperref[def:closed-under-countable-intersections]{Closed under countable intersections} &
Intersections of sequences of admitted subsets are admitted. \\
\hyperref[def:closed-under-arbitrary-unions]{Closed under arbitrary unions} &
Unions indexed by any set are admitted. \\
\bottomrule
\end{tabularx}
\end{center}
\end{toolkitbox}

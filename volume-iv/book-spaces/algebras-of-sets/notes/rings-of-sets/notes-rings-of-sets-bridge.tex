\begin{remark*}[Dictionary]
\[
\begin{array}{c|c}
\text{Set operation} & \text{Ring operation}\\ \hline
A\triangle B & A+B\\
A\cap B & A\cdot B\\
\varnothing & 0\\
X & 1
\end{array}
\]
Under this dictionary, a ring of sets corresponds to a Boolean rng, and an
algebra of sets corresponds to a Boolean ring with $1=X$.
\end{remark*}

\begin{remark*}[Structure theory lives in Volume VI]
For the algebraic structure theory -- idempotence implies characteristic $2$,
which implies $-p=p$, which implies commutativity, together with Stone
representation and the finite-order consequence that no Boolean ring has
order $3$ -- see Volume VI, Section~1.9, ``Boolean Rings.''
\end{remark*}

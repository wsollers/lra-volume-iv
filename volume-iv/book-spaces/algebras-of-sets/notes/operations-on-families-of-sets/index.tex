\section{Operations on Families of Sets}

\begin{toolkitbox}{Toolkit --- Operations on Families of Sets}
\small
\begin{tabular}{l p{0.62\linewidth}}
\toprule
\textbf{Name} & \textbf{Meaning} \\
\midrule
\hyperref[def:operation-on-members-of-family]{Operations on members} &
Use ordinary set operations with inputs drawn from a fixed family. \\
\hyperref[def:finite-operation-on-family]{Finite operations} &
Combine finitely many admitted subsets. \\
\hyperref[def:countable-operation-on-family]{Countable operations} &
Combine a sequence of admitted subsets. \\
\hyperref[def:arbitrary-operation-on-family]{Arbitrary operations} &
Combine an indexed family of admitted subsets with no countability bound. \\
\hyperref[def:finite-boolean-combination]{Finite Boolean combinations} &
Build subsets from finitely many inputs using union, intersection, and
complement. \\
\bottomrule
\end{tabular}
\end{toolkitbox}

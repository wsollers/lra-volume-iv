\begin{definitionbox}{Definition --- Normed Linear Space}
\begin{definition}[Normed Linear Space]\label{def:normed-linear-space}
Let \(V\) be a real vector space.  A \emph{normed linear space} is a pair
\((V,\|\cdot\|)\), where \(\|\cdot\|\) is a norm on \(V\).
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\text{\((V,\|\cdot\|)\) is a normed linear space}
\Longleftrightarrow
V\text{ is a real vector space}
\land
\operatorname{IsNorm}(V,\|\cdot\|).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\text{\((V,\|\cdot\|)\) is a normed linear space}
\]
means that the real vector space \(V\) is equipped with the norm
\(\|\cdot\|\).
\end{remark*}

\begin{remark*}[Interpretation]
A normed linear space is a vector space with a length structure.  The norm
measures the size of individual vectors and induces the metric
\(d(x,y)=\|x-y\|\) on the underlying set.
\end{remark*}

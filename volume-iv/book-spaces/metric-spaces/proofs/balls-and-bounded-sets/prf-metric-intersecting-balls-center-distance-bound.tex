\begin{theorem*}[Intersecting Balls Imply Center-Distance Bound]
Let \((X,d)\) be a metric space, let \(x,y\in X\), and let \(r,s>0\).  If
\[
  B_d(x;r)\cap B_d(y;s)\neq\varnothing,
\]
then \(d(x,y)<r+s\).
\end{theorem*}

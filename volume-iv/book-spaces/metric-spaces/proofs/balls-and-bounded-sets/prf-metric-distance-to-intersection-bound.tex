\begin{theorem*}[Distance to an Intersection Bound]
Let \((X,d)\) be a metric space, let \(x\in X\), and let
\(A,B\subseteq X\) have nonempty intersection.  Then
\[
  d(x,A\cap B)\geq \max\{d(x,A),d(x,B)\}.
\]
\end{theorem*}

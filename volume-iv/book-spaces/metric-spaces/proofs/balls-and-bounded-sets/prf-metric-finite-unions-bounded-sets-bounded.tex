\begin{theorem*}[Finite Unions of Bounded Sets Are Bounded]
Let \((X,d)\) be a metric space, let \(n\in\mathbb{N}\), and let
\(A_1,\ldots,A_n\subseteq X\).  If \(A_1,\ldots,A_n\) are bounded, then
\[
  \bigcup_{k=1}^n A_k
\]
is bounded.
\end{theorem*}

\begin{theorem*}[Distance to a Union]
Let \((X,d)\) be a metric space, let \(x\in X\), and let \(A,B\subseteq X\)
be nonempty.  Then
\[
  d(x,A\cup B)=\min\{d(x,A),d(x,B)\}.
\]
\end{theorem*}

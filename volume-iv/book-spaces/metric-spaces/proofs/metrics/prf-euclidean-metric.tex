\begin{corollary*}[Euclidean Metric]
For \(x,y\in\mathbb{R}^n\), define
\[
  d_2(x,y)=\sqrt{\sum_{i=1}^{n}|x_i-y_i|^2}.
\]
Then \((\mathbb{R}^n,d_2)\) is a metric space.
\end{corollary*}

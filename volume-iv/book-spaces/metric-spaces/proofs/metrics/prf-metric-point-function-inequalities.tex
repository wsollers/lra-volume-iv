\begin{theorem*}[Point Function Inequalities]
Let \((X,d)\) be a nonempty metric space and let \(z\in X\).  Then, for all
\(a,b\in X\),
\[
  \delta_z(b)-\delta_z(a)\leq d(a,b)\leq \delta_z(b)+\delta_z(a),
\]
and \(\delta_z(z)=0\).
\end{theorem*}

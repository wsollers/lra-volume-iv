\begin{theorem*}[A Norm Induces a Metric]
Let \(V\) be a real vector space and let \(\|\cdot\|\) be a norm on \(V\).
Define
\[
  d_{\|\cdot\|}(x,y):=\|x-y\|.
\]
Then \(d_{\|\cdot\|}\) is a metric on \(V\).
\end{theorem*}
